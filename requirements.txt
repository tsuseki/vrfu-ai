# Main pipeline extras — installed on top of ai-toolkit's requirements.txt by setup.bat.
# These are deps used by scripts/ and web/ but NOT pulled in by ai-toolkit itself.

# Prompt weighting + long-prompt support (used by generate.py and upscale.py via Compel)
compel>=2.3.1

# YAML parser used by every script (ai-toolkit also depends on it but pinning here too)
PyYAML>=6.0

# Image I/O for upscaler / web image serving (ai-toolkit also depends on it)
Pillow>=10.0

# Pin huggingface-hub to 0.x. transformers 4.57.x (pinned by ai-toolkit) requires
# huggingface-hub<1.0, but pip's resolver sometimes drags in 1.x anyway when
# other packages (peft, diffusers, accelerate) request a newer version. Locking
# it here as the last word so setup.bat ends up with a working combo.
huggingface-hub>=0.34.0,<1.0

# scripts/upscale.py so torchao 0.10.0 stays available for ai-toolkit's
# quantization features (FP8 training, lokr, etc.) without breaking our
# LoRA-only inference path.
